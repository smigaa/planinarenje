// ====== VALIDATION RULES ARRAY ======
const validationRules = [
  {
    id: "firstName",
    regex: /^[A-Z][a-zA-Z]{1,}$/,
    message: "Ime mora početi velikim slovom i imati makar 2 slova."
  },
  {
    id: "lastName",
    regex: /^[A-Z][a-zA-Z]{1,}$/,
    message: "Prezime mora početi velikim slovom i imati makar 2 slova."
  },
  {
    id: "email",
    regex: /^[^\s@]+@[^\s@]+\.com$/,
    message: "Email mora sadržati @ i završavati se sa .com."
  },
  {
    id: "password",
    regex: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[\W_]).{6,}$/,
    message: "Lozinka mora imati veliko, malo slovo, broj i specijalni karakter."
  }
];

// ====== LIVE VALIDATION ======
validationRules.forEach(rule => {
  const input = document.getElementById(rule.id);
  input.addEventListener("blur", () => validateField(input, rule));
});

function validateField(input, rule) {
  const errorDiv = input.nextElementSibling;
  if (!rule.regex.test(input.value.trim())) {
    input.classList.add("is-invalid");
    input.classList.remove("is-valid");
    errorDiv.textContent = rule.message;
    return false;
  } else {
    input.classList.remove("is-invalid");
    input.classList.add("is-valid");
    errorDiv.textContent = "";
    return true;
  }
}

// Confirm password
document.getElementById("confirmPassword").addEventListener("blur", function() {
  const pass = document.getElementById("password").value;
  const errorDiv = this.nextElementSibling;

  if (this.value !== pass || this.value === "") {
    this.classList.add("is-invalid");
    this.classList.remove("is-valid");
    errorDiv.textContent = "Lozinke se moraju poklapati.";
  } else {
    this.classList.remove("is-invalid");
    this.classList.add("is-valid");
    errorDiv.textContent = "";
  }
});

// ====== SUBMIT ======
document.getElementById("myForm").addEventListener("submit", function(e) {
  e.preventDefault();

  let isValid = true;

  validationRules.forEach(rule => {
    const input = document.getElementById(rule.id);
    if (!validateField(input, rule)) isValid = false;
  });

  // confirm password check
  const confirm = document.getElementById("confirmPassword");
  if (confirm.value !== document.getElementById("password").value || confirm.value === "") {
    confirm.classList.add("is-invalid");
    isValid = false;
  }

  // radio check
  const radio = document.querySelector('input[name="experience"]:checked');
  const radioError = document.querySelector('input[name="experience"]').parentElement.querySelector(".error-msg");
  if (!radio) {
    radioError.textContent = "Morate izabrati nivo iskustva.";
    isValid = false;
  } else {
    radioError.textContent = "";
  }

  // dropdown check
  if (select.value === "") {
    select.classList.add("is-invalid");
    select.nextElementSibling.textContent = "Morate izabrati planinu.";
    isValid = false;
  } else {
    select.classList.remove("is-invalid");
    select.classList.add("is-valid");
    select.nextElementSibling.textContent = "";
  }

  const formMessage = document.getElementById("formMessage");

  if (isValid) {
    formMessage.innerHTML = `<div class="success-msg">Hvala na prijavi!</div>`;
  } else {
    formMessage.innerHTML = `<div class="error-msg">Niste lepo popunili formu.</div>`;
  }

});
